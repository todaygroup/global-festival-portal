/* Banner sofort am Viewport – nicht auf window.load warten (Bilder/Embeds).
 * WICHTIG: Manager + Overlay müssen DIREKTE Kinder von <body> bleiben.
 * Manche Themes/Plugins (z. B. jPanelMenu) wickeln nach dem Laden alle
 * Body-Kinder in einen transformierten Wrapper. Ein transformierter Vorfahre
 * macht position:fixed relativ zu diesem Wrapper statt zum Viewport – das
 * Banner landet dann am Scroll-Ende und das Overlay deckt die ganze Höhe ab.
 * Daher hängen wir Banner/Overlay per MutationObserver immer wieder zurück
 * an <body>, falls sie verschoben wurden. */
(function () {
    var pos = "bottom";
    var modal = false;

    function detach() {
        var mgr = document.getElementById('gcms_cookie_manager');
        var ovl = document.getElementById('gcms_cookie_overlay');
        if (mgr && document.body && mgr.parentNode !== document.body) {
            document.body.appendChild(mgr);
        }
        if (ovl && document.body && ovl.parentNode !== document.body) {
            document.body.appendChild(ovl);
        }
    }
    window._CMS_cookie_detach = detach;

    detach();
    document.documentElement.classList.add('gcms_cookie_scroll_lock');
    document.body.classList.add('gcms_cookie_scroll_lock', 'gcms_cookie_open', 'gcms_cookie_body_v2_' + pos);
    if (modal) {
        document.body.classList.add('gcms_cookie_modal_active');
    }
    window.scrollTo(0, 0);

    /* Spätere Wrapper-Plugins abfangen: bei jeder Änderung an body wieder ausklinken.
     * Referenz global, damit _CMS_hide_cookie_einstellung den Observer beim
     * Schließen stoppen kann. */
    if (window.MutationObserver) {
        var mo = new MutationObserver(function () { detach(); });
        try { mo.observe(document.body, { childList: true }); window._CMS_cookieObserver = mo; } catch (e) { /* still */ }
    }
    /* Belt-and-suspenders für Plugins ohne sofortige DOM-Mutation */
    if (window.addEventListener) {
        window.addEventListener('load', detach, false);
        document.addEventListener('DOMContentLoaded', detach, false);
        setTimeout(detach, 300);
        setTimeout(detach, 1000);
    }
})();
</script>

    function _CMS_save_cookie_einstellung(reload) {
        var kat1 = document.getElementById("gcms_cookie_kat1_check").checked;
        var kat2 = document.getElementById("gcms_cookie_kat2_check").checked;
        var kat3 = document.getElementById("gcms_cookie_kat3_check").checked;

        if (!kat3) { _CMS_clear_local(3); }

        /* Kategorie abgewählt: Reload nötig, sonst bleiben per No-Reload aktivierte Embeds sichtbar */
        if (!reload && _CMS_consentReduced(kat1, kat2, kat3)) reload = 1;

        if (_COOKIE_UNLOAD) return;

        if ((kat2) && typeof GCMS_GA_run === 'function') GCMS_GA_run();

        var stat = 0;
        if (!reload && kat2) stat = 1;
        var gcms_java_enabled = navigator.javaEnabled() ? 1 : 0;
        var data = "kat1=" + kat1 + "&kat2=" + kat2 + "&kat3=" + kat3
                 + "&zust=0"
                 + "&im=17659"
                 + "&ds=17660"
                 + "&stat=" + stat
                 + "&refer=" + encodeURIComponent(_CMS_REF)
                 + "&orseid=17648"
                 + "&ink=0"
                 + "&st_res=" + screen.width + "x" + screen.height
                 + "&st_col=" + screen.colorDepth
                 + "&st_je=" + gcms_java_enabled
                 + "&loc=" + encodeURIComponent(window.location);

        var xhttp = new XMLHttpRequest();
        xhttp.open("POST", "/index.php?gcookiesave=1&spid=" + _CURRENT_SPID, true);
        xhttp.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
        xhttp.onreadystatechange = function () {
            if (this.readyState !== 4) return;
            if (this.status !== 200) return;
            _COOKIE_SAVED = true;

            if (reload) { document.location.reload(); return; }

            /* No-Reload-Pfad: Banner ausblenden, Overlay weg, Event feuern */
            _CMS_hide_cookie_einstellung(true);
            _CMS_update_cookie_erkl_zustand(!!kat1, !!kat2, !!kat3);

                });
                document.dispatchEvent(ev);
            } catch (e) { /* ältere Browser: still */ }
        };
        xhttp.send(data);
    }

(function(){
    "use strict";
    var form = document.querySelector('[data-such-form]');
    if (!form || form.dataset.acInit) { return; }
    form.dataset.acInit = "1";

    var API   = "\/de\/suche";
    var input = form.querySelector('[data-such-input]');
    var list  = form.querySelector('[data-such-list]');
    var live  = document.querySelector('[data-such-live]');
    var MIN   = 2;
    var activeIndex = -1;
    var current = [];
    var ctrl = null;

    function norm(s){
        return (s || '').toString().toLowerCase()
            .normalize('NFD').replace(/[\u0300-\u036f]/g, '')
            .replace(/\u00df/g, 'ss');
    }
    function esc(s){
        return (s || '').replace(/[&<>"]/g, function(c){
            return {'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;'}[c];
        });
    }

    function highlight(label, q){
        var nl = norm(label), nq = norm(q);
        var pos = nq ? nl.indexOf(nq) : -1;
        if (pos < 0) return esc(label);
        return esc(label.slice(0, pos)) + '<mark>' + esc(label.slice(pos, pos + q.length)) + '</mark>' + esc(label.slice(pos + q.length));
    }

    function apiUrl(q){
        return API + (API.indexOf('?') > -1 ? '&' : '?') + 'such_action=suggest&q=' + encodeURIComponent(q);
    }

    function close(){
        list.classList.remove('is-open');
        list.innerHTML = '';
        input.setAttribute('aria-expanded', 'false');
        activeIndex = -1;
        current = [];
    }

    function render(items, q){
        current = items || [];
        if (!current.length){ close(); return; }
        var html = '';
        for (var j = 0; j < current.length; j++){
            html += '<li id="ac-opt-' + j + '" class="autocomplete-item" role="option" aria-selected="false" tabindex="-1"><span class="ac_label">' + highlight(current[j].label, q) + '</span></li>';
        }
        list.innerHTML = html;
        list.classList.add('is-open');
        input.setAttribute('aria-expanded', 'true');
        activeIndex = -1;
        if (live) live.textContent = current.length + ' Vorschl\u00e4ge verf\u00fcgbar.';
    }

    var cache = {};            // Client-Cache pro Suchbegriff (Performance)
    function fetchSuggest(q){
        if (!API){ close(); return; }
        if (Object.prototype.hasOwnProperty.call(cache, q)){ render(cache[q], q); return; }
        if (ctrl){ try { ctrl.abort(); } catch(e){} }
        if (typeof AbortController !== 'undefined'){ ctrl = new AbortController(); }
        fetch(apiUrl(q), { credentials: 'same-origin', headers: { 'Accept': 'application/json' }, signal: ctrl ? ctrl.signal : undefined })
            .then(function(r){ return r.text(); })
            .then(function(txt){
                if ((input.value || '').trim() !== q) return;
                var d;
                try { d = JSON.parse(txt); }
                catch (e) {
                    if (window.console) console.warn('[Suche] Suggest-Antwort ist kein JSON:', apiUrl(q), txt.slice(0, 200));
                    close(); return;
                }
                var items = (d && d.items) || [];
                if (Object.keys(cache).length > 60) cache = {};   // Cache begrenzen
                cache[q] = items;
                render(items, q);
            })
            .catch(function(e){ if (e && e.name !== 'AbortError' && window.console) console.warn('[Suche] Suggest-Fetch fehlgeschlagen:', e); });
    }

    function choose(idx){
        var it = current[idx];
        if (!it) return;
        var u = it.url || '';
        // Sicherheit: nur sichere Ziele navigieren, sonst regulaere Suche ausloesen
        if (u && !/^\s*(javascript|data|vbscript):/i.test(u)){
            window.location.href = u;
        } else {
            input.value = it.label;
            form.submit();
        }
    }

    function setActive(idx){
        var items = list.querySelectorAll('li');
        items.forEach(function(li, i){
            var on = i === idx;
            li.classList.toggle('is-active', on);
            li.setAttribute('aria-selected', on ? 'true' : 'false');
            if (on && li.scrollIntoView){ li.scrollIntoView({ block: 'nearest' }); }
        });
        activeIndex = idx;
        input.setAttribute('aria-activedescendant', idx > -1 ? 'ac-opt-' + idx : '');
    }

    var debounce;
    input.addEventListener('input', function(){
        clearTimeout(debounce);
        var q = input.value.trim();
        debounce = setTimeout(function(){
            if (q.length < MIN){ close(); return; }
            fetchSuggest(q);
        }, 140);
    });

    input.addEventListener('keydown', function(e){
        if (!list.classList.contains('is-open')) return;
        var n = current.length;
        if (!n) return;
        if (e.key === 'ArrowDown'){ e.preventDefault(); setActive((activeIndex + 1) % n); }
        else if (e.key === 'ArrowUp'){ e.preventDefault(); setActive((activeIndex - 1 + n) % n); }
        else if (e.key === 'Home'){ e.preventDefault(); setActive(0); }
        else if (e.key === 'End'){ e.preventDefault(); setActive(n - 1); }
        else if (e.key === 'Enter'){ if (activeIndex > -1){ e.preventDefault(); choose(activeIndex); } }
        else if (e.key === 'Escape'){ e.preventDefault(); close(); }
        else if (e.key === 'Tab'){ close(); }
    });

    list.addEventListener('mousedown', function(e){
        var li = e.target.closest('li.autocomplete-item');
        if (!li) return;
        e.preventDefault();
        var items = Array.prototype.slice.call(list.querySelectorAll('li'));
        choose(items.indexOf(li));
    });

    document.addEventListener('click', function(e){
        if (!form.contains(e.target)) close();
    });
})();
